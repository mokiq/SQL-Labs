DROP DATABASE IF EXISTS e-books;
CREATE DATABASE IF NOT EXISTS e-books;
CREATE TABLE books(
    number INT(5) NOT NULL PRIMARY KEY,
    code INT(4) NOT NULL DEFAULT 0,
    new VARCHAR(3) NOT NULL DEFAULT 'No',
    title VARCHAR(150) NOT NULL DEFAULT '',
    price FLOAT(5, 2) NOT NULL CHECK (price > 0),
    publisher VARCHAR(50) NOT NULL,
    pages INT(11) DEFAULT NULL,
    format VARCHAR(20) DEFAULT NULL,
    date DATE DEFAULT NULL,
    tirage INT(11) DEFAULT NULL,
    topic VARCHAR(50) DEFAULT NULL,
    category VARCHAR(50) DEFAULT NULL
);

CREATE INDEX index_title ON
    e-books.books(title);
